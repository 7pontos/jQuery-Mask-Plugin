jQuery Mask Plugin

A jQuery Plugin for field masks.

Demos

Basic Examples

    $(document).ready(function(){
      $('.date').mask('11/11/1111');
      $('.time').mask('00:00:00');
      $('.date_time').mask('99/99/9999 00:00:00');
      $('.cep').mask('99999-999');
      $('.phone').mask('9999-9999');
      $('.phone_with_ddd').mask('(99) 9999-9999');
      $('.phone_us').mask('(999) 999-9999');
      $('.mixed').mask('AAA 000-S0S');
      $('.cpf').mask('999.999.999-99', {reverse: true});
      $('.money').mask('000.000.000.000.000,00', {reverse: true});
    });

Callback Examples

      var options =  { onComplete: function(cep) {
                        alert('Mask is done!:' + cep);
                      },
                      onKeyPress: function(cep, event, currentField, options){
                        console.log('An key was pressed!:', cep, ' event: ', event, 'currentField: ', currentField, ' options: ', options);
                      }};

      $('.cep_with_callback').mask('00000-000', options);

On-the-fly mask change

  var options =  {onKeyPress: function(cep){
                    var masks = ['00000-000', '0-00-00-00'];
                      mask = (cep.length>7) ? masks[1] : masks[0];
                    $('.crazy_cep').mask(mask, this);
                  }};

  $('.crazy_cep').mask('00000-000', options);

Removing the mask

    $('.date').data('mask').remove();

Mask Legend

  • 0,1,3,4,5,6,7,8 or 9: Only Numbers
  • A: Numbers and Letters
  • S: Only A-Z and a-z characters
