jQuery Mask Plugin

Demo

Usage Example

$(document).ready(function(){
  $('.date').mask('99/99/9999');
  $('.time').mask('99:99:99');
  $('.date_time').mask('99/99/9999 99:99:99');
  $('.cep').mask('99999-999');
  $('.phone').mask('9999-9999');
  $('.phone_with_ddd').mask('(99) 9999-9999');
  $('.phone_us').mask('(999) 999-9999');
});
